MACRO(add_file name)
  ADD_EXECUTABLE(${name} ${name}.cpp)
  TARGET_LINK_LIBRARIES(${name} ${OpenCV_LIBS} )
ENDMACRO()

MACRO(add_lib name)
  ADD_LIBRARY(${name} ${name}.h ${name}.cpp)
ENDMACRO()

MACRO(add_exec name)
  ADD_EXECUTABLE(${name} ${name}.cpp)
ENDMACRO()

MACRO(pack name)
  add_exec(${name})
  TARGET_LINK_LIBRARIES(${name}
                        commons
                        image_cv
                        ${OpenCV_LIBS} )
ENDMACRO()

add_lib(commons)
add_lib(image_cv)

pack(bitwise)
